pub struct Iter<'a, K: 'a, V: 'a>(_);

An iterator over the map's entries in ascending order.

Examples

Acquire through TreeMap::iter or the IntoIterator trait:

use tree::TreeMap;

let mut map = TreeMap::new();

map.insert(2, "b");
map.insert(1, "a");
map.insert(3, "c");

for (key, value) in &map {
    println!("{:?}: {:?}", key, value);
}
